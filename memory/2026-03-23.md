private MapNodeType DetermineNodeType(int floor)
{
    if ((floor + 1) % 5 == 0) return MapNodeType.Boss;      // 5,10,15层
    if ((floor + 1) % 5 == 4) return MapNodeType.Elite;     // 4,9,14层
    if ((floor + 1) % 5 == 3) return MapNodeType.Shop;      // 3,8,13层
    if ((floor + 1) % 5 == 2) return MapNodeType.Treasure;  // 2,7,12层
    return MapNodeType.Battle;                               // 其他层
}
